1. Write a function to find biggest of 2 numbers using 
a)if else

 #include <stdio.h>
 int large_number(int num1, int num2) 
{
    if (num1 > num2) 
      {
        return num1;
      }
    else 
      {
        return num2;
      }
}

int main() {
    int num1, num2;
    
    printf("Enter the first number: ");
    scanf("%d", &num1);
    
    printf("Enter the second number: ");
    scanf("%d", &num2);
    
    int large = large_number(num1, num2);
    
    printf("The biggest number is: %d\n", large);
    
    return 0;
)



b)ternary operator

#include <stdio.h>

int large_number(int num1, int num2)
{
    return (num1 > num2) ? num1 : num2;
}

int main() {
    int num1, num2;
    
    printf("Enter the first number: ");
    scanf("%d", &num1);
    
    printf("Enter the second number: ");
    scanf("%d", &num2);
    
    int large = large_number(num1, num2);
    
    printf("The biggest number is: %d\n", large);
    
    return 0;
}



2-Write a program to print the grade of the given student using if else if block

#include<stdio.h>
int main() {
    int marks;

    printf("Enter the marks of the student : ");
    scanf("%d", &marks);

    if (marks >= 90 && marks <= 100)
    {
        printf("Grade A\n");
    }
    else if (marks >= 75 && marks <= 89)
    {
        printf("Grade B\n");
    }
    else if (marks>= 60 && marks <= 74)
    {
        printf("Grade C\n");
    } 
    else if (marks >= 50 && marks <= 59)
    {
        printf("Grade D\n");
    } 
    else if (marks >= 0 && marks <= 49)
    {
        printf("Grade F\n");
    } 
    
}

#include <stdio.h>

int main() {
    float operand1, operand2, result;
    char operator;

    printf("Enter Number 1: ");
    scanf("%f", &operand1);

    printf("Enter the operator (+, -, *, /): ");
    scanf(" %c", &operator);
    
    printf("Enter Number 2: ");
    scanf("%f", &operand2);
    
    switch (operator) {
        case '+':
            result = operand1 + operand2;
            break;
        case '-':
            result = operand1 - operand2;
            break;
        case '*':
            result = operand1 * operand2;
            break;
        case '/':
            if (operand2 != 0) {
                result = operand1 / operand2;
            } else {
                printf("Error: Division by zero!\n");
                return 1; 
            }
            break;
        default:
            printf("Error: Invalid operator!\n");
            return 1; 
    }

    printf("Result: %.2f\n", result);

    return 0;
}
